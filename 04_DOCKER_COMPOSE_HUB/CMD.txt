# 이미지 생성
docker build -t [이미지명:태그명] [도커파일위치]

# 이미지 확인
docker images

# 이미지 삭제
docker rmi [id or 이미지명]

# 도커 이미지 내 진입
docker run -it  [id or 이미지:태그명] sh

# 임시 컨테이너 생성 후 외부포트와 연결
docker run -it --rm -p 3000:80 fn:1.1


# TAG 추가 [계정명/repository명]
docker tab bn rombird/rombird:bn
docker tab fn rombird/rombird:fn
docker tab redis rombird/rombird:redis
docker tab db rombird/rombird:db
