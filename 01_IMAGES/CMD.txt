# 이미지 생성
docker build -t [이미지명:태그명] 

# 이미지 삭제
docker rmi [id or 이미지명]

# 도커 이미지 내 진입
docker run -it [id or 이미지명] sh

# 임시 컨테이너 생성(종료시 삭제) + 포트 연결
docker run -it --rm -p 3000:80 fn


## BN
docker build -t bn:1.1 .
docker images # 확인
docker run -it bn:1.1 sh 
    /app에 들어와 있는 상태에서 
    # ls -l
    # java -jar app.jar -> springboot 실행
    # exit
docker run -it --rm -p 8080:8080 bn:1.1 # Dockerfile에 있는 외부포트번호와 일치시켜주기
